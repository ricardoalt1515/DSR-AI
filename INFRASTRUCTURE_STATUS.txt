   cd ../terraform/prod && terraform init && terraform apply

   cd backend && docker build ... && docker push ...

   cd ../infrastructure/terraform/prod && terraform apply

8. Verify Deployment (5 minutes)
   curl http://$(terraform output -raw alb_dns_name)/health
